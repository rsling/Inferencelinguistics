\section*{Symbols used in this book}

\begin{center}
  \resizebox{\textwidth}{!}{\begin{tabular}{llp{0.75\textwidth}}
    \toprule
    & \textbf{Name} & \textbf{Definition and Reference} \\
    \midrule
    % $.$ & \textit{.} & \\
    $n$ & \textit{sample size} & Number of observations\slash measurements in a sample \\
    $N$ & \textit{population size} & Mostly hypothetical number of objects in a population \\
    $p$ & \textit{p-value} & Frequentist pre-experiment probability of an obtained result under the Null \\
    $q$ & \textit{proportion} & Proportion of measurements with a specific value within a sample or population \\
    $s$ & \textit{variance} & Mean deviation of measurements from the mean in a sample \\
    $s^2$ & \textit{standard deviation} & Mean squared deviation of measurements from the mean in a sample \\
    \xsample & \textit{sample} & A sample of measurements as defined in the text in the form of a tuple (also called vector or array)\\
    \xmean & \textit{mean} & The (arithmetic) mean of the values in a sample \xsample \\
    \xmedian & \textit{median} & The middle value of a sorted sample \xsample \\
    \xmode & \textit{mode} & The most frequent value in \xsample (binary, nominal, ordinal) or the maximum of a distribution (numeric) \\
    \ysample & \textit{sample} & See \xsample, also for modifiers such as \ymean, \ymedian, and \ymode \\
    \zsample & \textit{sample} & See \xsample, also for modifiers such as \zmean, \zmedian, and \zmode \\
    \midrule
    $\mu$ & \textit{mean} & The same as \xmean, except for a population \\
    $\sigma$ & \textit{standard deviation} & The same as $s$ but for populations \\
    $\sigma^2$ & \textit{variance} & The same as $s^2$ but for populations \\
    \bottomrule
  \end{tabular}}
\end{center}
